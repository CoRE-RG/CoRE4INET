54 bool TTEScheduler::registerEvent(SchedulerEvent *event){
55 #ifdef DEBUG
56  Enter_Method("registerEvent(SchedulerEvent %s)",event->getName());
57 #else
58  Enter_Method_Silent();
59 #endif
60 
61  if (event->getKind() == ACTION_TIME_EVENT)
62  {
63  return registerEvent(dynamic_cast<SchedulerActionTimeEvent*>(event), false);
64  }
65  else if (event->getKind() == TIMER_EVENT)
66  {
67  take(event);
68  registredEvents.push_back(event);
69  SchedulerTimerEvent *timerEvent = (SchedulerTimerEvent*) event;
70  scheduleAt(simTime() + currentTick * timerEvent->getTimer(), event);
71  }
72  return true;
73 }

76 bool TTEScheduler::registerEvent(SchedulerActionTimeEvent *actionTimeEvent, bool forceNextCycle)
77 {
78 #ifdef DEBUG
79  Enter_Method("registerEvent(SchedulerEvent %s, forceNextCycle %d)",event->getName(), forceNextCycle);
80 #else
81  Enter_Method_Silent();
82 #endif
83 
84  take(actionTimeEvent);
85  registredEvents.push_back(actionTimeEvent);
86 
87  //Check whether event is in cycle
88  if(actionTimeEvent->getAction_time()>cycleTicks){
89  bubble("Schedule contains out of cycle events!");
90  return false;
91  }
92 
93  if ((actionTimeEvent->getAction_time() <= getTicks()) || forceNextCycle)
94  {
95  scheduleAt(lastCycleStart
96  + currentTick
97  * (actionTimeEvent->getAction_time() + cycleTicks),
98  actionTimeEvent);
99  }
100  else
101  {
102  scheduleAt(lastCycleStart + currentTick * (actionTimeEvent->getAction_time()),
103  actionTimeEvent);
104  }
105  return true;
106 }

115 void TTEScheduler::handleMessage(cMessage *msg)
116 {
117  if (msg->isSelfMessage() && (msg->getKind() == ACTION_TIME_EVENT || msg->getKind() == TIMER_EVENT))
118  {
119  SchedulerEvent *event = (SchedulerEvent*) msg;
120  registredEvents.remove(event);
121  sendDirect(event, event->getDestinationGate());
122  }
123  else if (msg->isSelfMessage() && msg->getKind() == NEW_CYCLE)
124  {
125  //First the precision is changed for the next cycle
126  changeDrift();
127 
128  emit(newCycle, 1L);
129  cycles++;
130  lastCycleStart = simTime();
131  lastNewCycleMessage = simTime();
132  lastCycleTicks += cycleTicks;
133  scheduleAt(lastCycleStart + currentTick * cycleTicks, msg);
134  newCyclemsg = msg;
135 
136  correctEvents();
137  }
138 }

140 void TTEScheduler::changeDrift(){
141  simtime_t newTick = currentTick+SimTime(par("drift_change").doubleValue());
142  if((newTick-tick)>maxDrift)
143  par("current_tick").setDoubleValue((tick+maxDrift).dbl());
144  else if((newTick-tick)<-maxDrift)
145  par("current_tick").setDoubleValue((tick-maxDrift).dbl());
146  else
147  par("current_tick").setDoubleValue(newTick.dbl());
148  emit(currentDrift, par("current_tick").doubleValue()-tick);
149 }

151 void TTEScheduler::correctEvents(){
152  for(std::list<SchedulerEvent*>::const_iterator registredEvent = registredEvents.begin(); registredEvent != registredEvents.end(); registredEvent++){
153  if((*registredEvent)->getKind() == ACTION_TIME_EVENT){
154  SchedulerActionTimeEvent *actionTimeEvent = (SchedulerActionTimeEvent*)*registredEvent;
155  cancelEvent(actionTimeEvent);
156  //TODO lots of improvements!
157  if (actionTimeEvent->getAction_time() > getTicks())
158  {
159  scheduleAt(lastCycleStart + currentTick * actionTimeEvent->getAction_time(),
160  actionTimeEvent);
161  }
162  else if (actionTimeEvent->getAction_time() == getTicks())
163  {
164  scheduleAt(simTime(), actionTimeEvent);
165  }
166  else
167  {
168  if(lastCycleStart>simTime()){
169  scheduleAt(lastCycleStart
170  + currentTick
171  * (actionTimeEvent->getAction_time()),
172  actionTimeEvent);
173  }
174  else{
175  scheduleAt(lastCycleStart
176  + currentTick
177  * (actionTimeEvent->getAction_time() + cycleTicks),
178  actionTimeEvent);
179  }
180  }
181  }
182  }

192 void TTEScheduler::clockCorrection(int32_t ticks){
193  Enter_Method("clock correction %d ticks",ticks);
194 
195  lastCycleStart+=ticks*currentTick;
196 
197  //Now correct the new cylce time
198  cancelEvent(newCyclemsg);
199  scheduleAt(simTime()-(simTime()-lastCycleStart) + currentTick * cycleTicks, newCyclemsg);
200  //Now correct the events

204 uint32_t TTEScheduler::getTicks()
205 {
206  if(simTime() >= lastCycleStart){
207  return floor((simTime() - lastCycleStart) / currentTick);
208  }
209  else{
210  return cycleTicks - floor((lastCycleStart - simTime()) / currentTick);
