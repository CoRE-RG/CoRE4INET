57 void RCBuffer::handleMessage(cMessage *msg)
58 {
59  CTBuffer::handleMessage(msg);
60 
61  if(destinationGates.size() > 0)
62  {
63  if (msg->arrivedOn("in"))
64  {
65  if (bagExpired)
66  {
67  cMessage *outgoingMessage = getFrame();
68  if(outgoingMessage){
69  bagExpired = false;
70  numReset = 0;
71  //Send Message
72  for (std::list<cGate*>::iterator dgate = destinationGates.begin(); dgate != destinationGates.end(); ++dgate)
73  {
74  sendDirect(outgoingMessage->dup(),0,0, *dgate);
75  }
76  if(gate("out")->isConnected()){
77  send(outgoingMessage->dup(),"out");
78  }
79  recordPacketSent();
80  delete outgoingMessage;
81  }
82  }
83  }
84  else if (msg->arrivedOn("schedulerIn") && msg->getKind() == TIMER_EVENT)
85  {
86  cMessage *outgoingMessage = getFrame();
87  if (outgoingMessage)
88  {
89  bagExpired = false;
90  numReset = 0;
91  //Send Message
92  for (std::list<cGate*>::iterator gate = destinationGates.begin(); gate != destinationGates.end(); ++gate)
93  {
94  sendDirect(outgoingMessage->dup(), *gate);
95  }
96  recordPacketSent();
97  delete outgoingMessage;
98  }
99  else
100  {
101  bagExpired = true;
102  getDisplayString().setTagArg("i2", 0, "");
103  }
104  }
105  }
106 }

114 void RCBuffer::resetBag()
115 {
116  Enter_Method("resetBag()");
117  //This is the moment when the message was transmitted execute transmit callbacks if there are some
118  for(std::map<TTEApplicationBase*,Callback*>::const_iterator iter = transmitCallbacks.begin();
119  iter != transmitCallbacks.end(); ++iter){
120  iter->first->executeCallback(iter->second);
121  }
122 
123  //Set icon:
124  getDisplayString().setTagArg("i2", 0, "status/hourglass");
125 
126  numReset++;
127  if (numReset == destinationGates.size())
128  {
129  //Reregister scheduler
