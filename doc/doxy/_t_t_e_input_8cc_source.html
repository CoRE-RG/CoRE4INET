45 void TTEInput::handleMessage(cMessage *msg)
46 {
47  if (msg->arrivedOn("in"))
48  {
49  EtherFrame *frame = (EtherFrame*) msg;
50 
51  int i = msg->findPar("received");
52  cMsgPar* par;
53  if( i >=0 )
54  par = &msg->par(i);
55  else
56  par = &msg->addPar("received");
57  par->setLongValue(((TTEScheduler*)getParentModule()->getParentModule()->getSubmodule("tteScheduler"))->getTicks());
58 
59  i = msg->findPar("received_total");
60  if( i >=0 )
61  par = &msg->par(i);
62  else
63  par = &msg->addPar("received_total");
64  par->setLongValue(((TTEScheduler*)getParentModule()->getParentModule()->getSubmodule("tteScheduler"))->getTotalTicks());
65 
66 
67  i = msg->findPar("received_port");
68  if( i >=0 )
69  par = &msg->par(i);
70  else
71  par = &msg->addPar("received_port");
72  par->setLongValue(getParentModule()->getIndex());
73 
74  //Auf CTCs verteilen oder BE traffic
75  if (isCT(frame))
76  {
77  std::map<uint16_t, std::list<Incoming *> >::iterator incomingList = incomings.find(getCTID(frame));
78  if (incomingList != incomings.end())
79  {
80  //Send to all CTCs for the CT-ID
81  for (std::list<Incoming*>::iterator incoming = incomingList->second.begin(); incoming
82  != incomingList->second.end(); incoming++)
83  {
84  sendDirect(frame->dup(), (*incoming)->gate("in"));
85  }
86  delete frame;
87  }
88  else
89  {
90  emit(ctDroppedSignal, 1);
91  hadError=true;
92  if(ev.isGUI()){
93  bubble("No matching buffer configured");
94  getDisplayString().setTagArg("i2", 0, "status/excl3");
95  getDisplayString().setTagArg("tt", 0, "WARNING: Input configuration problem - No matching buffer configured");
96  getParentModule()->getDisplayString().setTagArg("i2", 0, "status/excl3");
97  getParentModule()->getDisplayString().setTagArg("tt", 0, "WARNING: Input configuration problem - No matching buffer configured");
98  getParentModule()->getParentModule()->getDisplayString().setTagArg("i2", 0, "status/excl3");
99  getParentModule()->getParentModule()->getDisplayString().setTagArg("tt", 0, "WARNING: Input configuration problem - No matching buffer configured");
100  }
101  delete frame;
102  }
103  }
104  //Sonst BE
105  else
106  {
107  if(promiscuous || frame->getDest().isMulticast()){
108  send(msg, "out");
109  }
110  else{
111  MACAddress address;
112  address.setAddress(frame->getArrivalGate()->getPathStartGate()->getOwnerModule()->par("address"));
113  if(frame->getDest().equals(address)){
114  send(msg, "out");
115  }
116  else{
117  delete msg;
118  }
119  }
120  }
121  }
122 }

124 void TTEInput::handleParameterChange(const char* parname){
125  ctMask = (uint32_t)par("ct_mask").longValue();
126  ctMarker = (uint32_t)par("ct_marker").longValue();
127  promiscuous = par("promiscuous").boolValue();
128 
129  incomings.clear();
130 
131  std::string incomingsString = par("incomings").stdstringValue();
132  std::vector<std::string> incomingPaths;
133  split(incomingsString,',',incomingPaths);
134  for(std::vector<std::string>::iterator incomingPath = incomingPaths.begin();
135  incomingPath!=incomingPaths.end();incomingPath++){
136  cModule* module = simulation.getModuleByPath((*incomingPath).c_str());
137  if(!module){
138  module = findModuleWhereverInNode((*incomingPath).c_str(),this);
139  }
140  if(module){
141  if(findContainingNode(module)!=findContainingNode(this)){
142  opp_error("Configuration problem of incomings: Module: %s is not in node %s! Maybe a copy-paste problem?", (*incomingPath).c_str(),
143  findContainingNode(this)->getFullName());
144  }
145  else
146  {
147  Incoming *incoming = dynamic_cast<Incoming*> (module);
148  if(incoming){
149  Buffer *buffer = dynamic_cast<Buffer*> (incoming->gate("out")->getPathEndGate()->getOwner());
150  if(buffer && buffer->hasPar("ct_id")){
151  incomings[buffer->par("ct_id").longValue()].push_back(incoming);
152  }
153  else{
154  opp_error("Incoming module %s has no Buffer attached with ct_id configured!", (*incomingPath).c_str());
155  }
156  }
157  else{
158  opp_error("Configuration problem of tt_buffers: Module: %s is no Incoming module!", (*incomingPath).c_str());
159  }
160  }
161  }
162  else{
163  opp_error("Configuration problem of incomings: Module: %s could not be resolved!", (*incomingPath).c_str());
164  }
165  }
166 }

168 bool TTEInput::isCT(EtherFrame *frame)
169 {
170  unsigned char macBytes[6];
171  frame->getDest().getAddressBytes(macBytes);
172  //Check for ct
173  if ((((macBytes[0] << 24) | (macBytes[1] << 16) | (macBytes[2] << 8) | (macBytes[3])) & ctMask) == (ctMarker
174  & ctMask))
175  {
176  return true;
177  }
178  //TODO More checking ?
179  return false;
180 }
