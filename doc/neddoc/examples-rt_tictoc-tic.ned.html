package ttethernet.examples.rt_tictoc;

import ttethernet.nodes.ethernet.TTEEtherHost;
import ttethernet.ctc.TTIncoming;
import ttethernet.buffer.TTDoubleBuffer;
import ttethernet.ctc.RCIncoming;
import ttethernet.buffer.RCQueueBuffer;

//
// tic: configuration in tic.ini.
//
// - Sends time-triggered messages at 1ms with VL-ID 100, receives rate-constrained messages with VL-ID 101
//
module Tic extends TTEEtherHost
{
    @display("bgb=503,314");
    submodules:
        vl_100_ctc: TTIncoming {
            parameters:
                @display("p=350,60");

        }
        vl_100: TTDoubleBuffer {
            parameters:
                @display("p=450,60");
        }
        vl_101_ctc: RCIncoming {
            parameters:
                @display("p=350,160");

        }
        vl_101: RCQueueBuffer {
            parameters:
                @display("p=450,160");
        }
    connections:
        vl_100_ctc.out --> vl_100.in;
        vl_101_ctc.out --> vl_101.in;
}
